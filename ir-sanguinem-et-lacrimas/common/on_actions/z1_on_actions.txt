on_actions = {

  on_declare_war = {
    effect = {
      if = {
        limit = {
          ROOT = {
            has_country_flag = uses_chemical_weapons
          }
        }
        country_event = { id = chemical_warfare.5 days = 2}
      }
    }
  }

  on_startup = {
		effect = {
      every_country = {
        set_country_flag = cw_hesitation_flag
        set_global_flag = { flag = CW_global_chem_users value = 1 }
      }
      every_country = {
        limit = {
          is_league_of_nations_1936 = yes
        }
        set_country_flag = LN_league_member
      }
		}
	}

}
