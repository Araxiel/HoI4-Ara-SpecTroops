add_namespace = chemical_warfare

##chemical_warfare started
country_event = {
	id = chemical_warfare.1
	title = chemical_warfare.1.t
	desc = chemical_warfare.1.d
	picture = GFX_report_event_gas_ger

	fire_only_once = yes
  is_triggered_only = yes

	option = {
		name = chemical_warfare.1.a

		news_event = { hours = 6 id = chemical_warfare.4 }

		every_country = {
			limit = {
				is_neighbor_of = ROOT
				NOT = { is_in_faction_with = ROOT }
			}
			add_opinion_modifier = { target = ROOT modifier = chemical_warfare_started }
		}
		every_country = {
			limit = {
				has_war_with = ROOT
			}
			add_opinion_modifier = { target = ROOT modifier = chemical_warfare_started_war }
		}
		if = {
			limit = {
					has_government = democratic
			}
			add_national_unity = -0.02
		}

		set_country_flag = uses_chemical_weapons
		ROOT = { save_global_event_target_as = chemical_warfare_starter }
		set_global_flag = { flag = nations_using_chem_warfare value = 1 }
	}

}

## chemical warfare development started
country_event = {
	id = chemical_warfare.6
	title = chemical_warfare.6.t
	picture = GFX_report_event_gas_ukf

	desc = chemical_warfare.6.d

	mean_time_to_happen = {
		months = 4
		modifier = {
			factor = 0.7
			has_global_flag = {
				flag = chemical_weapons_development_countries
				value > 0
			}
		}
		modifier = {
			factor = 0.7
			has_global_flag = {
				flag = chemical_weapons_development_countries
				value > 2
			}
		}
		modifier = {
			factor = 0.7
			has_global_flag = {
				flag = chemical_weapons_development_countries
				value > 4
			}
		}
		modifier = {
			factor = 0.7
			threat > 35
		}
	}

	trigger = {
		NOT = { has_global_flag = chemical_warfare_activated }
		has_global_flag = {
			flag = chemical_weapons_allowed_countries
			value < 5
		}
		any_country = {
			NOT = { is_ai = yes }
		}
	}

	immediate = {

		random_country = {
			limit = {
				NOT = { has_country_flag = chem_wep_developer }
				is_ai = yes
				is_major = yes
			}
			save_event_target_as = chem_wep_developer
		}

		random_list = {
			65 = {
				random_country = {
					limit = {
						is_major = yes
						OR = {
							has_government = communist
							has_government = fascist
						}
						NOT = { has_country_flag = chem_wep_developer }
						is_ai = yes
					}
					# effect
					save_event_target_as = chem_wep_developer
				}
			}
			15 = {
				random_country = {
					limit = {
						is_major = yes
						has_government = democratic
						NOT = { has_country_flag = chem_wep_developer }
						is_ai = yes
					}
					# effect
					save_event_target_as = chem_wep_developer
				}
			}
			10 = {
				random_country = {
					limit = {
						is_major = yes
						NOT = { has_country_flag = chem_wep_developer }
						is_ai = yes
					}
					# effect
					save_event_target_as = chem_wep_developer
				}
			}
			10 = {
				random_country = {
					limit = {
						NOT = { has_country_flag = chem_wep_developer }
						is_ai = yes
					}
					save_event_target_as = chem_wep_developer
				}
			}
		}

		event_target:chem_wep_developer = {
			set_country_flag = { flag = chem_wep_developer days = 600 }
		}

		modify_global_flag = { flag = chemical_weapons_development_countries value = 1 }
	}

	# we don't care
	option = {
		name = chemical_warfare.6.a

		custom_effect_tooltip = chemical_warfare.6.a.tt

		hidden_effect = {
			event_target:chem_wep_developer = {
				add_tech_bonus = {
					name = dd_bonus
					bonus = 0.25
					uses = 1
					category = chemical_warfare
				}
			}

			modify_global_flag = { flag = chemical_weapons_allowed_countries value = 1 }
		}

	}

	# how dare they
	option = {
		name = chemical_warfare.6.b

		custom_effect_tooltip = chemical_warfare.6.b.tt

		event_target:chem_wep_developer = {
			add_timed_idea = { idea = chem_reprimanded days = 400 }
			add_opinion_modifier = { target = ROOT modifier = reprimanded_for_chem_weapons }
		}

	}

	# call convention
	option = {
		name = chemical_warfare.6.c

		trigger = {
			has_global_flag = {
				flag = chemical_weapons_development_countries
				value < 4
			}
		}

		custom_effect_tooltip = chemical_warfare.6.c.tt
		hidden_effect = {
			news_event = { days = 3 id = chemical_warfare.7 }
		}
		country_event = { months = 1 id = chemical_warfare.8 }

	}

}
